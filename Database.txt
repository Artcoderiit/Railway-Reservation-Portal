CREATE TABLE admin(
fname varchar(40) NOT NULL,
lname varchar(40) NOT NULL,
uname varchar(40) PRIMARY KEY,
password varchar(40) NOT NULL,
) COMMENT 'this table represents admin and interacts with tables train and stations';

CREATE TABLE booking_agent(
fname varchar(40) NOT NULL,
lname varchar(40) NOT NULL,
uname varchar(40) PRIMARY KEY,
password varchar(40) NOT NULL,
gender char(1) NOT NULL,
dob timestamp NOT NULL,
email varchar(40) NOT NULL,
phone varchar(40) NOT NULL,
creditcard varchar(16) NOT NULL,
address varchar(100) NOT NULL
);

CREATE TABLE trains(
trainno int PRIMARY KEY,
source varchar(40) NOT NULL,
dest varchar(40) NOT NULL,
ac_coaches int NOT NULL,
sl_coaches int NOT NULL,

);

CREATE TABLE stations(
stationid int PRIMARY KEY,
stationname varchar(40) NOT NULL,


);

CREATE TABLE ticket(
pnr varchar(40) PRIMARY KEY,
bookedby varchar(40) NOT NULL,
no_of_seats int NOT NULL,
doj timestamp NOT NULL,
coach char(2) NOT NULL,
trainno int NOT NULL,
status int NOT NULL,
amount int NOT NULL,
foreign key (trainno) references trains(trainno) ON UPDATE CASCADE ON DELETE CASCADE
foreign key (bookedby) references booking_agent(uname) ON UPDATE CASCADE ON DELETE CASCADE
);




CREATE TABLE passenger(
pid varchar(40) PRIMARY KEY,
fname varchar(40) NOT NULL,
lname varchar(40) NOT NULL,
age int NOT NULL,
gender char(1) NOT NULL,
seatno int NOT NULL,
bookedby varchar(40) NOT NULL,
berth char(2) NOT NULL,
pnr varchar(40) NOT NULL,
foreign key (bookedby) references booking_agent(uname) ON UPDATE CASCADE ON DELETE CASCADE,
foreign key (pnr) references ticket(pnr) ON UPDATE CASCADE ON DELETE CASCADE
);
